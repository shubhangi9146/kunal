this is dev file.. import java.util.Random; import java.util.Scanner; public class CricketGame { public static void main(String[] args) { Scanner scanner = new Scanner(System.in); Random random = new Random(); System.out.println("Welcome to the Cricket Game!"); int targetScore = random.nextInt(101) + 100; // Generating a target score between 100 and 200 int totalScore = 0; int balls = 0; System.out.println("Target Score: " + targetScore); while (balls < 6) { System.out.print("\nEnter runs scored for ball " + (balls + 1) + ": "); int runs = scanner.nextInt(); if (runs < 0 || runs > 6) { System.out.println("Invalid input. Please enter a number between 0 and 6."); continue; } int outcome = random.nextInt(7); // Simulating the outcome of the ball (0 to 6) System.out.println("Outcome of the ball: " + outcome); if (runs == outcome) { System.out.println("Out! You scored " + totalScore + " runs in " + (balls + 1) + " balls."); break; } totalScore += runs; balls++; System.out.println("Current Score: " + totalScore + " runs in " + balls + " balls."); if (totalScore > targetScore) { System.out.println("Congratulations! You've chased down the target!"); break; } } if (balls == 6 && totalScore < targetScore) { System.out.println("Sorry! You couldn't chase down the target. Your final score: " + totalScore); } scanner.close(); } }
